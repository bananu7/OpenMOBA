  4 function direction(a, b) {
  5 	"use strict";
  6 	var dx,
  7 		dz;
  8 
  9 	dx = b.x - a.x;
 10 	dz = b.z - a.z;
 11 	return Math.atan2(dz, dx);

 14 function distance(a, b) {
 15 	"use strict";
 16 	var dx,
 17 		dz;
 18 
 19 	dx = b.x - a.x;
 20 	dz = b.z - a.z;
 21 	return Math.sqrt(dx * dx + dz * dz);

 24 /**
 25  * @class
 26  */
 27 var objectManager = (function () {
 28 	"use strict";
 29 	/** @ignore */
 30 	var C;
 31 
 32 	/** @private */
 33 	var objects = {};
 34 	/** @private */
 35 	var scene;
 36 	/** @private */
 37 	var toBeLoaded = 0;
 38 	/** @private */
 39 	var loaded = 0;
 40 
 41 	/**
 42 	 * @exports C as objectManager
 43 	 */
 44 	C = {
 45 		/**
 46 		 * Default THREE JSON loader
 47 		 * @type {THREE}
 48 		 */
 49 		loaderJSON: new THREE.JSONLoader(),
 50 
 51 		/**
 52 		 * Sets scene for the objects to be added to
 53 		 * @param {THREE.Scene} sc Scene
 54 		 */
 55 		setScene: function (sc) {
 56 			scene = sc;
 57 		},
 58 
 59 		/**
 60 		 * Adds object to the manager and tries to load it. When it's loaded, executes callback cb.
 61 		 * @param {String}   name Object's identificator
 62 		 * @param {String}   path Path to object's model to be loaded
 63 		 * @param {Function} cb   Callback executed after the model is loaded with this object as an argument.
 64 		 */
 65 		addObject: function (name, path, cb) {
 66 			if (!scene) {
 67 				throw new Error("objectManager: Scene was not defined!");
 68 			}
 69 			if (!objects[name]) {
 70 				++toBeLoaded;
 71 				objects[name] = true; // loading
 72 				this.loaderJSON.load(path, function (geometry, materials) {
 73 					objects[name] = new THREE.Mesh(geometry, new THREE.MeshFaceMaterial(materials));
 74 					scene.add(objects[name]);
 75 					++loaded;
 76 					if (cb && typeof cb === "function") {
 77 						cb(objects[name]);
 78 					}
 79 					if (toBeLoaded === loaded) {
 80 						C.onAllLoaded(true);
 81 					}
 82 				});
 83 			} else {
 84 				throw new Error("objectManager: Object of this name already exists!");
 85 			}
 86 		},
 87 
 88 		/**
 89 		 * 
 90 		 * @param  {String} name Object's identificator
 91 		 * @return {THREE.Mesh}      Reference to the object
 92 		 */
 93 		getByName: function (name) {
 94 			return objects[name];
 95 		},
 96 
 97 		/**
 98 		 * Removed object of given identificator from scene and from object manager
 99 		 * @param  {String} name Object's identificator
100 		 */
101 		removeObject: function (name) {
102 			if (!scene) {
103 				throw new Error("objectManager: Scene was not defined!");
104 			}
105 			if (objects[name]) {
106 				scene.remove(objects[name]);
107 				delete objects[name];
108 			}
109 		},
110 
111 		/**
112 		 * Pushes back callback functions which are to be executed when all models are loaded.
113 		 * @param  {Function} cb Callback function to be executed with all objects as an argument
114 		 * @function
115 		 */
116 		onAllLoaded: (function () {
117 			var cbs = [];
118 			return function (cb) {
119 				if (!cb) {
120 					throw new Error("objectManager: No callback given!");
121 				}
122 				if (typeof cb === "function") {
123 					cbs.push(cb);
124 				}
125 				if (loaded && loaded === toBeLoaded) {
126 					for (var i = 0; i < cbs.length; ++i) {
127 						cbs[i](objects);
128 					}
129 					cbs.length = 0;
130 				}
131 			};
132 		}())
133 	};
134 
135 	return C;
136 }());
